import {Component, Input} from '@angular/core';

@Component({
    selector: 'app-govuk-table',
    templateUrl: './govuk-table.component.html',
    styleUrls: ['./govuk-table.component.scss']
})
export class GovukTableComponent {

    @Input() classes = '';

    @Input() caption = 'Dates and amounts';
    @Input() firstCellIsHeader = true;
    @Input() head = [
        { text: 'Date' },
        { text: 'Amount' }
    ];
    @Input() rows = [
        [
            { text: 'First 6 weeks' },
            { text: '£109.80 per week' }
        ],
        [
            { text: 'Next 33 weeks' },
            { text: '£109.80 per week' }
        ],
        [
            { text: 'Total estimated pay'
            },
            { text: '£4,282.20' }
        ]
    ];

    constructor() { }

}

<table class="{{'govuk-table ' + classes}}">
    <caption class="govuk-table__caption">{{caption}}</caption>
    <thead class="govuk-table__head">
    <tr class="govuk-table__row">
        <th class="govuk-table__header" scope="col" *ngFor="let h of head">{{h.text}}</th>
    </tr>
    </thead>
    <tbody class="govuk-table__body">
    <tr data-selector="table-row" class="govuk-table__row" *ngFor="let r of rows">
        <ng-container *ngFor="let col of r; first as isFirst">
            <th data-selector="table-header" class="govuk-table__header" scope="row"  *ngIf="isFirst && firstCellIsHeader">{{col.text}}</th>
            <td data-selector="table-cell" class="govuk-table__cell"  *ngIf="!(isFirst && firstCellIsHeader)">{{col.text}}</td>
        </ng-container>
    </tr>
    </tbody>
</table>
